--USER 생성
CREATE USER JSP2 IDENTIFIED BY JSP2;
--권한부여
GRANT RESOURCE, CONNECT TO JSP2;
--테이블스페이스 지정
ALTER USER JSP2 DEFAULT TABLESPACE USERS QUOTA UNLIMITED ON USERS;

--테이블 생성
CREATE TABLE USERS2(
    ID VARCHAR2(30),
    PW VARCHAR2(30) NOT NULL,
    NAME VARCHAR2(30) NOT NULL,
    EMAIL VARCHAR2(30),
    GENDER CHAR(1),
    REGDATE DATE DEFAULT SYSDATE
);

--PK,CHECK KEY 설정
ALTER TABLE USERS2 ADD CONSTRAINT USERS2_PK PRIMARY KEY (ID);
ALTER TABLE USERS2 ADD CONSTRAINT USERS2_CHECK CHECK (GENDER IN('M', 'F'));

--테이블 생성 확인
SELECT * FROM USERS2;
